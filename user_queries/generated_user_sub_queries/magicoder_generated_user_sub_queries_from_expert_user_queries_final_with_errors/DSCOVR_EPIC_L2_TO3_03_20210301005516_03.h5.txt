import h5py

# Open the file in read mode
with h5py.File('DSCOVR_EPIC_L2_TO3_03_20210301005516_03.h5', 'r') as f:
    # Accessing datasets and attributes
    ozone = f['/Ozone'][:]
    latitude = f['/Latitude'][:]
    longitude = f['/Longitude'][:]
    
    # Setting attributes
    units = f['/Ozone'].attrs.get('units')
    long_name = f['/Ozone'].attrs.get('long_name')

import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap, midpoint
from cartopy.mpl.gridliner import LONGITUDE_FORMATTER, LATITUDE_FORMATTER

# Calculate the midpoint of the latitude and longitude arrays
mid_lat = np.mean(latitude)
mid_lon = np.mean(longitude)

# Create an orthographic projection centered at the calculated midpoint
m = Basemap(projection='ortho', lat_0=mid_lat, lon_0=mid_lon)

# Plot the ozone data as a scatter plot
x, y = m(longitude, latitude)
scatter = plt.scatter(x, y, c=ozone, s=1, cmap='jet')

# Add map features
m.drawcoastlines()
m.drawparallels(np.arange(-90., 120., 30.), labels=[True, False, False, True], fontsize=10)
m.drawmeridians(np.arange(0., 420., 60.), labels=[False, False, False, True], fontsize=10)

# Create a colorbar with a width of 0.022 and padding of 0.01
