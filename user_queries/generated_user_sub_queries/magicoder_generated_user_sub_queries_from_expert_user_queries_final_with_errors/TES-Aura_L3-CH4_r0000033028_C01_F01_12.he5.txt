import h5py

# Open File
file = h5py.File('TES-Aura_L3-CH4_r0000033028_C01_F01_12.he5', 'r')

# Retrieve Data
grid_info = file['/HDFEOS INFORMATION/StructMetadata.0'][()].decode('utf-8')
SurfacePressure = file['HDFEOS/GRIDS/NadirGrid/Data Fields/SurfacePressure'][:]
metadata = {'_FillValue': SurfacePressure.attrs.get('_FillValue'), 
            'MissingValue': SurfacePressure.attrs.get('MissingValue'), 
            'Title': SurfacePressure.attrs.get('Title').decode(), 
            'Units': SurfacePressure.attrs.get('Units').decode()}

# Process Data
invalid_values = np.logical_or(SurfacePressure == metadata['_FillValue'], SurfacePressure == metadata['MissingValue'])
SurfacePressure[invalid_values] = np.nan
masked_array = np.ma.masked_array(SurfacePressure, np.isnan(SurfacePressure))

# Retrieve Coordinates
Longitude = file['/HDFEOS/GRIDS/NadirGrid/Data Fields/Longitude'][:]
Latitude = file['/HDFEOS/GRIDS/NadirGrid/Data Fields/Latitude'][:]
