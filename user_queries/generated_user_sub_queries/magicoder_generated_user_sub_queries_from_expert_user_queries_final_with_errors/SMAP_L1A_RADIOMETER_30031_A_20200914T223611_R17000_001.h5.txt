import h5py

def read_hdf5(filename):
    with h5py.File(filename, 'r') as f:
        # Accessing datasets and attributes
        ds = f['HighResolution_Moments_Data']['moments16_right_ascension']
        attrs = ds.attrs
        
        # Extracting specific attributes
        units = attrs['units'] if 'units' in attrs else None
        long_name = attrs['long_name'] if 'long_name' in attrs else None
        _FillValue = attrs['_FillValue'] if '_FillValue' in attrs else None
        valid_max = attrs['valid_max'] if 'valid_max' in attrs else None
        valid_min = attrs['valid_min'] if 'valid_min' in attrs else None
        
    return units, long_name, _FillValue, valid_max, valid_min
