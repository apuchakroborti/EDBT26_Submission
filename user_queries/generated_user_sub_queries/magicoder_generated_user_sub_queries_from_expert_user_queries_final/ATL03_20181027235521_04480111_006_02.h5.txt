```python
import os
import h5py

FILE_NAME = "ATL03_20181027235521_04480111_006_02.h5"

def access_datasets(file_name):
    with h5py.File(file_name, 'r') as f:
        lat = f['/gt1l/geolocation/reference_photon_lat'][:]
        lon = f['/gt1l/geolocation/reference_photon_lon'][:]
        data = f['/gt1l/geolocation/tx_pulse_energy'][:]
        
        _FillValue = data.attrs['_FillValue'][0]
        units = data.attrs['units'].decode('ascii', 'replace')
        long_name = data.attrs['long_name'].decode('ascii', 'replace')
    
    return lat, lon, data, _FillValue, units, long_name
```

```python
import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap

def plot_map(lat, lon, data):
    map = Basemap(projection='cyl', llcrnrlat=-90, urcrnrlat=90, llcrnrlon=-180, urcrnrlon=180, resolution='l')
    
    map.drawcoastlines(linewidth=0.5)
    map.drawparallels([-90, -60, 0, 60, 90], labels=[True, False, False, True])
    map.drawmeridians([-180, -120, 0, 120, 180], labels=[False, False, False, True])
    
    x, y = map(lon, lat)
    plt.scatter(x, y, c=data, s=0.1, cmap='jet', edgecolors='none', linewidths=0)
    plt.colorbar(label=f'{units}')
    plt.title(os.path.basename(FILE_NAME) + ' - ' + long_name)
    
    plt.show()
```
