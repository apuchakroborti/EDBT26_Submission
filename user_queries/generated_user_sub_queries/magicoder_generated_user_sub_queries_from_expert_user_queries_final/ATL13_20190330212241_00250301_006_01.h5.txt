import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap

# Create a new figure window
fig = plt.figure()

# Add a map to the figure
map = Basemap(projection='cyl', llcrnrlat=-90, urcrnrlat=90, llcrnrlon=-180, urcrnrlon=180, resolution='l')

# Plot coastlines on the map
map.drawcoastlines(linewidth=0.5)

# Add parallels and meridians to the map
parallels = np.arange(-90., 90, 30.)
meridians = np.arange(-180., 180., 45.)
map.drawparallels(parallels, labels=[True, False, False, False], linewidth=0.5)
map.drawmeridians(meridians, labels=[False, False, False, True], linewidth=0.5)

# Plot data on the map
x, y = map(data['longitude'], data['latitude'])
plt.scatter(x, y, c=data['data'], s=0.1, cmap='jet', edgecolors='none', linewidths=0)

# Add colorbar and title to the figure
cbar = plt.colorbar()
