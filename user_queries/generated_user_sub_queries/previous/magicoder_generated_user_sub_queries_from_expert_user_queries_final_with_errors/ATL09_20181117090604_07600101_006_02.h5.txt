import h5py
FILE_NAME = 'ATL09_20181117090604_07600101_006_02.h5'
with h5py.File(FILE_NAME, 'r') as f:
    latitude = f['/profile_1/high_rate']['latitude'][()]
    longitude = f['/profile_1/high_rate']['longitude'][()]
    apparent_surf_reflec = f['/profile_1/high_rate']['apparent_surf_reflec'][()]
    units = f['/profile_1/high_rate']['apparent_surf_reflec'].attrs['units'].decode('ascii', 'replace')
    long_name = f['/profile_1/high_rate']['apparent_surf_reflec'].attrs['long_name'].decode('ascii', 'replace')

import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap
m = Basemap(projection='cyl', llcrnrlat=-90, urcrnrlat=90, llcrnrlon=-180, urcrnrlon=180)
