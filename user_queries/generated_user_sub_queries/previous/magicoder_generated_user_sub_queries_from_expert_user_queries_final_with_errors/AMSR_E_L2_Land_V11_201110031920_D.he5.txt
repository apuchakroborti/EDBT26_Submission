# Open the file in read-only mode.
with h5py.File('AMSR_E_L2_Land_V11_201110031920_D.he5', 'r') as f:
    # Retrieve main data from the attribute 'SoilMoistureSCA' at `AMSR-E Level 2 Land Data/Data/Combined NPD and SCA Output Fields`.
    soil_moisture = f['AMSR-E Level 2 Land Data/Data/Combined NPD and SCA Output Fields'].attrs['SoilMoistureSCA']
    
    # Extract latitude data, longitude data from the group `/HDFEOS/POINTS/AMSR-E Level 2 Land Data/Data/Combined NPD and SCA Output Fields`.

import numpy as np
