 HDF5 Dataset Access Sub-query:
```python
import h5py
FILE_NAME = "ATL12_20190330212241_00250301_006_03.h5"
with h5py.File(FILE_NAME, 'r') as f:
    # Accessing datasets and attributes
    latitude = f['/gpsr/navigation']['latitude']
    longitude = f['/gpsr/navigation']['longitude']
    bg_cnt_50shot_s = f['/atlas/pce1/background']['bg_cnt_50shot_s']
    delta_time = f['/atlas/pce1/background']['delta_time']
    
    # Extracting attributes
    unit = bg_cnt_50shot_s.attrs['unit'].decode('ascii', 'replace')
    long_name = bg_cnt_50shot_s.attrs['long_name'].decode('ascii', 'replace')
```

from mpl_toolkits.basemap import Basemap

# First Plot (Time vs. Counts)
fig = plt.figure(figsize=(10, 10))
