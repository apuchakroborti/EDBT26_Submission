```python
import h5py

# Open an HDF5 file.
with h5py.File('filename.h5', 'r') as f:
    # Read datasets: soil_moisture from `soil_moisture`, 
    # latitude from `latitude`, and longitude from `longitude` from the group `Soil_Moisture_Rettrieval_Data_AM`.
    soil_moisture = f['Soil_Moisture_Retrieval_Data_AM']['soil_moisture'][:]
    latitude = f['Soil_Moisture_Retrieval_Data_AM']['latitude'][:]
    longitude = f['Soil_Moisture_Retrieval_Data_AM']['longitude'][:]
    
    # Retrieve attributes: units, long_name, _FillValue, valid_max, and valid_min.
    units = f['Soil_Moisture_Retrieval_Data_AM'].attrs['units']
    long_name = f['Soil_Moisture_Retrieval_Data_AM'].attrs['long_name']
    _FillValue = f['Soil_Moisture_Retrieval_Data_AM']['soil_moisture'].attrs['_FillValue']
    valid_max = f['Soil_Moisture_Retrieval_Data_AM']['soil_moisture'].attrs['valid_max']
    valid_min = f['Soil_Moisture_Retrieval_Data_AM']['soil_moisture'].attrs['valid_min']
```

```python
# Decode units and long name.
units = units.decode('ascii', 'replace')
long_name = long_name.decode('ascii', 'replace')

# Calculate invalid data.
invalid_data = np.logical_or(soil_moisture < valid_min, soil_moisture > valid_max)
soil_moisture[invalid_data] = _FillValue
soil_moisture[np.isnan(soil_moisture)] = np.nan
```

```python
import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap

# Set up an equidistant cylindrical projection map using Basemap with low resolution.
map = Basemap(projection='cyl', llcrnrlat=-90, urcrnrlat=90, llcrnrlon=-180, urcrnrlon=180, resolution='l')

# Draw coastlines with line width 0.5, parallels at [-180, -135, -90, -45, 0, 45, 90], and meridians at [-180, -135, -90, -45, 0, 45, 90, 135, 180].
map.drawcoastlines(linewidth=0.5)
map.drawparallels([-180, -135, -90, -45, 0, 45, 90], labels=[True, False, False, True])
map.drawmeridians([-180, -135, -90, -45, 0, 45, 90, 135, 180], labels=[False, False, False, True])

# Create a scatter plot of soil_moisture data using longitude and latitude coordinates.
x, y = map(longitude, latitude)
map.scatter(x, y, s=1, c=soil_moisture, cmap='jet', edgecolor='none', linewidth=0)

# Add a colorbar labeled with the decoded units.
cbar = plt.colorbar(label=units)

# Set the title as the HDF5 file's basename concatenated with the long_name attribute.
```
