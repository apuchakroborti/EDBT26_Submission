```python
import h5py
FILE_NAME = 'MOP02J-20131129-L2V19.9.3.he5'
with h5py.File(FILE_NAME, 'r') as f:
    dataset = f['/HDFEOS/SWATHS/MOP02/Data Fields/RetrievedSurfaceTemperature']
    data = dataset[...]
    units = dataset.attrs['units'].decode()
    _FillValue = dataset.attrs['_FillValue']
```
