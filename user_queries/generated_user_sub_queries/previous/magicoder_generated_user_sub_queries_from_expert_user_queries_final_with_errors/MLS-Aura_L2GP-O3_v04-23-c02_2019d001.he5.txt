```python
import h5py

# Open the file in read mode
with h5py.File('MLS-Aura_L2GP-O3_v04-23-c02_2019d001.he5', 'r') as f:
    # Accessing datasets and attributes
    ozone = f['/HDFEOS/SWATHS/O3/Data Fields/L2gpValue'][0:70, :]
    latitudes = f['/HDFEOS/SWATHS/O3/Geolocation Fields/Latitude'][0:70]
    pressures = f['/HDFEOS/SWATHS/O3/Geolocation Fields/Pressure'][:55]
    
    # Extracting units and title information
    ozone_units = f['/HDFEOS/SWATHS/O3/Data Fields/L2gpValue'].attrs.get('units')
    latitudes_title = f['/HDFEOS/SWATHS/O3/Geolocation Fields/Latitude'].attrs.get('long_name')
    pressures_units = f['/HDFEOS/SWATHS/O3/Geolocation Fields/Pressure'].attrs.get('units')
```

```python
import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap

# Create a pcolormesh plot with parameters: latitude, longitude, and ozone data transposed (data.T)
m = Basemap(projection='cyl')
x, y = m(*np.meshgrid(latitudes, pressures))
plt.pcolormesh(x, y, ozone.T, cmap=plt.cm.jet)

# Set x-axis label using latitude title and units, and y-axis label using longitude title and units
m.drawparallels(np.arange(-90.,120.,30.), labels=[True,False,False,True], fontsize=10)
plt.xlabel('%s (%s)' % (latitudes_title, 'degrees'))
m.drawmeridians(np.arange(0.,420.,60.), labels=[True,False,False,True], fontsize=10)
plt.ylabel('%s (%s)' % ('Pressure', pressures_units))

# Set the plot title to include the data title concatenated with start and end times
start_time = '1993-01-01' + str(int(np.round((70 / ozone.shape[1]) * (f['Time'].value - 1))))
end_time = '1993-01-01' + str(int(np.round((70 / ozone.shape[1]) * f['Time'].value)))
plt.title('%s: %s to %s' % (f['/HDFEOS/SWATHS/O3/Data Fields/L2gpValue'].attrs.get('long_name'), start_time, end_time))

# Configure the plot
plt.gca().invert_yaxis()
plt.yscale('log')
plt.gca().xaxis.set_major_formatter(plt.FuncFormatter(lambda x, pos: '%.1f' % x))

# Save the plot at the location of the data file and Display the plot
```
