```python
import h5py
FILE_NAME = 'MOP02N-20180311-L2V19.9.2.he5'
with h5py.File(FILE_NAME, 'r') as f:
    dset = f['/HDFEOS/SWATHS/MOP02/Data Fields/RetrievedCOTotalColumn']
    data = dset[...]
    units = dset.attrs['units']
    _FillValue = dset.attrs['_FillValue']
    data[_FillValue] = np.nan  # replace _FillValue with NaN
```
NumPy Data Preprocessing Sub-query:
```python
masked_data = np.ma.masked_where(data == _FillValue, data)
min_val = np.nanmin(masked_data)
max_val = np.nanmax(masked_data)
filtered_data = (masked_data - min_val) / (max_val - min_val)  # normalize the data
```
Plotting and Visualization Sub-query:
```python
import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap
f = h5py.File(FILE_NAME, 'r')

longitude = f['/HDFEOS/SWATHS/MOP02/Geolocation Fields/Longitude'][...]
map = Basemap(projection='cyl', resolution='c', llcrnrlat=min(latitude), urcrnrlat=max(latitude), llcrnrlon=min(longitude), urcrnrlon=max(longitude))
map.drawcoastlines(linewidth=0.5)
map.drawparallels(np.arange(-90, 91, 45), labels=[True, False, False, True])
map.drawmeridians(np.arange(-180, 181, 45), labels=[False, False, False, True])
```
