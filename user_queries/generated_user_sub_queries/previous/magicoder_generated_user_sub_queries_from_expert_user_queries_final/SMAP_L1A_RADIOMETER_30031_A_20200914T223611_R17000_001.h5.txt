import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap

def run(filename):
    with h5py.File(filename, 'r') as f:
        ds = f['/HighResolution_Moments_Data/moments16_right_ascension']
        lat = f['/HighResolution_Moments_Data/moments16_lat'][:]
        lon = f['/HighResolution_Moments_Data/moments16_lon'][:]
        
    fig, ax = plt.subplots()
    
    m = Basemap(projection='cyl', llcrnrlat=-90, urcrnrlat=90, llcrnrlon=-180, urcrnrlon=180)
    m.drawcoastlines()
    m.drawparallels(np.arange(-90., 90., 30.), labels=[1, 0, 0, 0])
    m.drawmeridians(np.arange(-180., 180., 60.), labels=[0, 0, 0, 1])
    
    x, y = m(lon, lat)
    cs = m.scatter(x, y, c=ds[:], s=40, cmap='jet')
    plt.colorbar(cs, label=ds.attrs["long_name"])
