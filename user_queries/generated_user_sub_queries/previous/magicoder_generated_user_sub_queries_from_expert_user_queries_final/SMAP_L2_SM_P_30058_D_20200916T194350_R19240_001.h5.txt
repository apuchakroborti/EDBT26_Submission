```python
import numpy as np

# Assuming 'data' is the numpy array you want to preprocess
_FillValue = data.attrs['_FillValue']
valid_max = data.attrs['valid_max']
valid_min = data.attrs['valid_min']

mask = np.logical_or(data >= valid_max, data <= valid_min)
data = np.where(mask, _FillValue, data)
```
