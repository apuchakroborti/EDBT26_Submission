```python
import os, datetime
import h5py
import numpy as np
from matplotlib import pyplot as plt
from mpl_toolkits.basemap import Basemap
from matplotlib.ticker import FormatStrFormatter, ScalarFormatter

# 1. HDF5 Dataset Access Sub-query:
with h5py.File('HIRDLS-Aura_L2_v07-00-20-c01_2008d077.he5', 'r') as f:
    # Accessing datasets and attributes
    o3 = f['/HDFEOS/SWATHS/HIRDLS/Data Fields/O3'][0, :]
    pressure = f['/HDFEOS/SWATHS/HIRDLS/Geolocation Fields/Pressure'][:]
    time = f['/HDFEOS/SWATHS/HIRDLS/Geolocation Fields/Time'][0]
    
    # Reading attributes
    o3_units = f[o3.attrs['Units']][()].decode()
    pressure_units = f[pressure.attrs['Units']][()].decode()
    time_units = f[time.attrs['Units']][()].decode()
    
    # Decoding attribute values
    o3_title = f[o3.attrs['Title']][()].decode()
    pressure_title = f[pressure.attrs['Title']][()].decode()
    time_title = f[time.attrs['Title']][()].decode()
    
    _FillValue = f[o3.attrs['_FillValue']][()]

# 2. NumPy Data Preprocessing Sub-query:
o3[o3 == _FillValue] = np.nan
datestr = datetime.datetime(1993, 1, 1) + datetime.timedelta(seconds=time[0])

# 3. Plotting and Visualization Sub-query:
fig, ax = plt.subplots()
m = Basemap(projection='cyl')
x, y = m(*np.meshgrid(pressure[:], o3))
cs = m.scatter(x, y, c=o3, edgecolors='k', linewidths=0.5)
fig.colorbar(cs, label=f'{o3_title} [{o3_units}]')
ax.set_yscale('log')
ax.invert_yaxis()
ax.xaxis.set_major_formatter(FormatStrFormatter('%g'))
ax.yaxis.set_major_formatter(ScalarFormatter())
plt.title(f'{os.path.basename("HIRDLS-Aura_L2_v07-00-20-c01_2008d077.he5")} - {o3_title} [{o3_units}]')
plt.show()
```

This code reads an HDF5 file, accesses datasets and attributes, decodes attribute values, replaces a specific value with NaN, calculates a date from the time data, and finally plots O3 data against pressure using a log scale.
