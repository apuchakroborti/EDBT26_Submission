```python
import os, re, h5py

def read_hdf5(file):
    with h5py.File(file, 'r') as f:
        d_lat = f['/Data_1HZ/Geolocation/d_lat'][:]
        d_lon = f['/Data_1HZ/Geolocation/d_lon'][:]
        d_surface_temp = f['/Data_1HZ/Atmosphere/d_Surface_temp'][:]
        d_UTCTime_1 = f['/Data_1HZ/Time/d_UTCTime_1'][:]
    return d_lat, d_lon, d_surface_temp, d_UTCTime_1
```

```python
import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap

def plot(lat, lon, temp):
    fig = plt.figure(figsize=(15, 6))
    
    ax1 = fig.add_subplot(121)
    m = Basemap(projection='cyl', llcrnrlat=-90, urcrnrlat=90, llcrnrlon=-180, urcrnrlon=180, resolution='l')
    x, y = m(lon, lat)
    sc = ax1.scatter(x, y, c=temp, s=0.1, cmap='jet', edgecolors='none')
    plt.colorbar(sc)
```
