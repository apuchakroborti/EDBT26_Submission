 HDF5 Dataset Access Sub-query:
```python
import h5py

# Open the HDF5 file
with h5py.File('ozone_data.h5', 'r') as f:
    # Extract the 3D ProfileOzone data and its attributes
    profile_ozone = f['/Data_Fields/ProfileOzone']
    units, long_name = profile_ozone.attrs['units'], profile_ozone.attrs['long_name']
    
    # Extract the Latitude data and its attributes
    latitude = f['/Data_Fields/Latitude']
    lat_units, lat_long_name = latitude.attrs['units'], latitude.attrs['long_name']
    
    # Extract ProfilePressureLevels data and its attributes
    pressure_levels = f['/Data_Fields/ProfilePressureLevels']
    pl_units, pl_long_name = pressure_levels.attrs['units'], pressure_levels.attrs['long_name']
    
    # Extract Date data and its attributes
    date = f['/Data_Fields/Date']
    date_units, date_long_name = date.attrs['units'], date.attrs['long_name']
```

import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap

# Create a figure and set up the map
fig, ax = plt.subplots()
m = Basemap(projection='merc', llcrnrlat=min(latitude), urcrnrlat=max(latitude), 
            llcrnrlon=min(longitude), urcrnrlon=max(longitude))

# Plot the data
x, y = m(longitude, latitude)
m.contourf(x, y, profile_ozone[0, :, :], 15, cmap=plt.cm.jet)

# Add colorbar and labels
cbar = m.colorbar(location='bottom', pad="10%")
cbar.set_label('Ozone Units')
ax.set_xlabel(f'{lat_long_name} ({lat_units})')
